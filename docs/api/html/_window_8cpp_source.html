23bool Window::Create(const WindowDesc& desc) {
24 // Enable per-monitor DPI awareness V2
25 SetProcessDpiAwarenessContext(DPI_AWARENESS_CONTEXT_PER_MONITOR_AWARE_V2);
26
27 // Register window class
28 WNDCLASSEXW wc = {};
29 wc.cbSize = sizeof(WNDCLASSEXW);
30 wc.style = CS_HREDRAW | CS_VREDRAW;
31 wc.lpfnWndProc = WindowProcStatic;
32 wc.hInstance = GetModuleHandle(nullptr);
33 wc.hCursor = LoadCursor(nullptr, IDC_ARROW);
34 wc.hbrBackground = (HBRUSH)GetStockObject(BLACK_BRUSH);
35 wc.lpszClassName = WINDOW_CLASS_NAME;
36
37 if (!RegisterClassExW(&wc)) {
38 return false;
39 }
40
41 // Calculate window size with DPI
42 UpdateDPI();
43
44 int scaledWidth = static_cast<int>(desc.width * m_dpiScale);
45 int scaledHeight = static_cast<int>(desc.height * m_dpiScale);
46
47 // Window style
48 DWORD style = WS_OVERLAPPEDWINDOW;
49 if (!desc.resizable) {
50 style &= ~(WS_THICKFRAME | WS_MAXIMIZEBOX);
51 }
52
53 // Adjust window rect to account for borders
54 RECT rect = {0, 0, scaledWidth, scaledHeight};
55 AdjustWindowRect(&rect, style, FALSE);
56
57 // Create window
58 m_hwnd = CreateWindowExW(
59 0,
60 WINDOW_CLASS_NAME,
61 desc.title.c_str(),
62 style,
63 CW_USEDEFAULT, CW_USEDEFAULT,
64 rect.right - rect.left,
65 rect.bottom - rect.top,
66 nullptr,
67 nullptr,
68 GetModuleHandle(nullptr),
69 this // Pass this pointer for WM_CREATE
70 );
71
72 if (!m_hwnd) {
73 return false;
74 }
75
76 m_width = desc.width;
77 m_height = desc.height;
78
79 return true;
80}

153LRESULT Window::WindowProc(HWND hwnd, UINT msg, WPARAM wParam, LPARAM lParam) {
154 switch (msg) {
155 case WM_CLOSE:
156 if (OnClose) {
157 OnClose();
158 }
159 // Don't call DestroyWindow here - let Application::Shutdown() handle
160 // cleanup in the correct order (DX12 resources before window)
161 PostQuitMessage(0);
162 return 0;
163
164 case WM_DESTROY:
165 PostQuitMessage(0);
166 return 0;
167
168 case WM_SIZE:
169 {
170 int width = LOWORD(lParam);
171 int height = HIWORD(lParam);
172
173 if (wParam == SIZE_MINIMIZED) {
174 // Don't resize when minimized
175 } else {
176 HandleResize(width, height);
177 }
178 }
179 return 0;
180
181 case WM_ENTERSIZEMOVE:
182 m_isResizing = true;
183 return 0;
184
185 case WM_EXITSIZEMOVE:
186 m_isResizing = false;
187 // Trigger final resize
188 {
189 RECT rect;
190 if (GetClientRect(hwnd, &rect)) {
191 HandleResize(rect.right - rect.left, rect.bottom - rect.top);
192 }
193 }
194 return 0;
195
196 case WM_SIZING:
197 // During live resize, get the new size and force redraw
198 if (lParam) {
199 RECT* dragRect = reinterpret_cast<RECT*>(lParam);
200
201 // Calculate frame size by comparing current window and client rects
202 RECT windowRect, clientRect;
203 GetWindowRect(hwnd, &windowRect);
204 GetClientRect(hwnd, &clientRect);
205
206 int frameWidth = (windowRect.right - windowRect.left) - (clientRect.right - clientRect.left);
207 int frameHeight = (windowRect.bottom - windowRect.top) - (clientRect.bottom - clientRect.top);
208
209 // Calculate new client size from proposed window rect
210 int newWidth = (dragRect->right - dragRect->left) - frameWidth;
211 int newHeight = (dragRect->bottom - dragRect->top) - frameHeight;
212
213 if (newWidth > 0 && newHeight > 0) {
214 HandleResize(newWidth, newHeight);
215 // Force immediate redraw
216 if (OnPaint) {
217 OnPaint();
218 }
219 }
220 }
221 return TRUE;
222
223 case WM_PAINT:
224 {
225 PAINTSTRUCT ps;
226 BeginPaint(hwnd, &ps);
227
228 if (OnPaint) {
229 OnPaint();
230 }
231
232 EndPaint(hwnd, &ps);
233 }
234 return 0;
235
236 case WM_KEYDOWN:
237 case WM_SYSKEYDOWN:
238 if (OnKeyEvent) {
239 OnKeyEvent(static_cast<UINT>(wParam), true);
240 }
241 return 0;
242
243 case WM_KEYUP:
244 case WM_SYSKEYUP:
245 if (OnKeyEvent) {
246 OnKeyEvent(static_cast<UINT>(wParam), false);
247 }
248 return 0;
249
250 case WM_CHAR:
251 if (OnCharEvent) {
252 OnCharEvent(static_cast<wchar_t>(wParam));
253 }
254 return 0;
255
256 case WM_LBUTTONDOWN:
257 case WM_RBUTTONDOWN:
258 case WM_MBUTTONDOWN:
259 if (OnMouseButton) {
260 int x = GET_X_LPARAM(lParam);
261 int y = GET_Y_LPARAM(lParam);
262 int button = (msg == WM_LBUTTONDOWN) ? 0 : (msg == WM_RBUTTONDOWN) ? 1 : 2;
263 OnMouseButton(x, y, button, true);
264 }
265 return 0;
266
267 case WM_LBUTTONUP:
268 case WM_RBUTTONUP:
269 case WM_MBUTTONUP:
270 if (OnMouseButton) {
271 int x = GET_X_LPARAM(lParam);
272 int y = GET_Y_LPARAM(lParam);
273 int button = (msg == WM_LBUTTONUP) ? 0 : (msg == WM_RBUTTONUP) ? 1 : 2;
274 OnMouseButton(x, y, button, false);
275 }
276 return 0;
277
278 case WM_MOUSEMOVE:
279 if (OnMouseMove) {
280 int x = GET_X_LPARAM(lParam);
281 int y = GET_Y_LPARAM(lParam);
282 OnMouseMove(x, y);
283 }
284 return 0;
285
286 case WM_MOUSEWHEEL:
287 if (OnMouseWheel) {
288 int delta = GET_WHEEL_DELTA_WPARAM(wParam);
289 OnMouseWheel(delta);
290 }
291 return 0;
292
293 case WM_DPICHANGED:
294 {
295 UpdateDPI();
296
297 // Windows suggests a new size for the window
298 RECT* rect = reinterpret_cast<RECT*>(lParam);
299 SetWindowPos(hwnd, nullptr,
300 rect->left, rect->top,
301 rect->right - rect->left,
302 rect->bottom - rect->top,
303 SWP_NOZORDER | SWP_NOACTIVATE);
304 }
305 return 0;
306 }
307
308 return DefWindowProcW(hwnd, msg, wParam, lParam);
309}

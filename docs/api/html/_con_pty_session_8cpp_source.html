26bool ConPtySession::Start(const std::wstring& commandline, int cols, int rows) {
27 if (m_running) {
28 spdlog::warn("ConPTY session already running");
29 return false;
30 }
31
32 m_cols = cols;
33 m_rows = rows;
34
35 // Create pipes for ConPTY communication
36 SECURITY_ATTRIBUTES sa = { sizeof(SECURITY_ATTRIBUTES), nullptr, TRUE };
37
38 if (!CreatePipe(&m_hPipePtyIn, &m_hPipeIn, &sa, 0)) {
39 spdlog::error("Failed to create input pipe: {}", GetLastError());
40 return false;
41 }
42
43 if (!CreatePipe(&m_hPipeOut, &m_hPipePtyOut, &sa, 0)) {
44 spdlog::error("Failed to create output pipe: {}", GetLastError());
45 CloseHandle(m_hPipePtyIn);
46 CloseHandle(m_hPipeIn);
47 return false;
48 }
49
50 // Make sure the handles we keep are not inherited
51 SetHandleInformation(m_hPipeIn, HANDLE_FLAG_INHERIT, 0);
52 SetHandleInformation(m_hPipeOut, HANDLE_FLAG_INHERIT, 0);
53
54 // Create the pseudoconsole
55 COORD consoleSize = { static_cast<SHORT>(cols), static_cast<SHORT>(rows) };
56 HRESULT hr = CreatePseudoConsole(consoleSize, m_hPipePtyIn, m_hPipePtyOut, 0, &m_hPC);
57
58 if (FAILED(hr)) {
59 spdlog::error("Failed to create pseudoconsole: 0x{:08X}", hr);
60 CloseHandle(m_hPipePtyIn);
61 CloseHandle(m_hPipePtyOut);
62 CloseHandle(m_hPipeIn);
63 CloseHandle(m_hPipeOut);
64 return false;
65 }
66
67 // Close the ConPTY ends of the pipes (ConPTY owns them now)
68 CloseHandle(m_hPipePtyIn);
69 CloseHandle(m_hPipePtyOut);
70 m_hPipePtyIn = INVALID_HANDLE_VALUE;
71 m_hPipePtyOut = INVALID_HANDLE_VALUE;

73 // Prepare startup info with pseudoconsole attribute
74 STARTUPINFOEXW siEx = {};
75 siEx.StartupInfo.cb = sizeof(STARTUPINFOEXW);
76 siEx.StartupInfo.dwFlags = STARTF_USESTDHANDLES;
77 siEx.StartupInfo.hStdInput = nullptr;
78 siEx.StartupInfo.hStdOutput = nullptr;
79 siEx.StartupInfo.hStdError = nullptr;

81 // Allocate attribute list
82 SIZE_T attrListSize = 0;
83 InitializeProcThreadAttributeList(nullptr, 1, 0, &attrListSize);
84
85 siEx.lpAttributeList = reinterpret_cast<LPPROC_THREAD_ATTRIBUTE_LIST>(malloc(attrListSize));
86 if (!siEx.lpAttributeList) {
87 spdlog::error("Failed to allocate attribute list");
88 ClosePseudoConsole(m_hPC);
89 CloseHandle(m_hPipeIn);
90 CloseHandle(m_hPipeOut);
91 return false;
92 }
93
94 if (!InitializeProcThreadAttributeList(siEx.lpAttributeList, 1, 0, &attrListSize)) {
95 spdlog::error("Failed to initialize attribute list: {}", GetLastError());
96 free(siEx.lpAttributeList);
97 ClosePseudoConsole(m_hPC);
98 CloseHandle(m_hPipeIn);
99 CloseHandle(m_hPipeOut);
100 return false;
101 }
102
103 // Set the pseudoconsole attribute
104 if (!UpdateProcThreadAttribute(siEx.lpAttributeList, 0, PROC_THREAD_ATTRIBUTE_PSEUDOCONSOLE,
105 m_hPC, sizeof(HPCON), nullptr, nullptr)) {
106 spdlog::error("Failed to update proc thread attribute: {}", GetLastError());
107 DeleteProcThreadAttributeList(siEx.lpAttributeList);
108 free(siEx.lpAttributeList);
109 ClosePseudoConsole(m_hPC);
110 CloseHandle(m_hPipeIn);
111 CloseHandle(m_hPipeOut);
112 return false;
113 }
114
115 // Create the process
116 PROCESS_INFORMATION pi = {};
117 std::wstring cmdline = commandline; // CreateProcessW needs non-const
118
119 BOOL success = CreateProcessW(
120 nullptr, // Application name
121 cmdline.data(), // Command line
122 nullptr, // Process security attributes
123 nullptr, // Thread security attributes
124 FALSE, // Inherit handles
125 EXTENDED_STARTUPINFO_PRESENT, // Creation flags
126 nullptr, // Inherit environment
127 nullptr, // Current directory
128 &siEx.StartupInfo, // Startup info
129 &pi // Process information
130 );
131
132 // Cleanup attribute list
133 DeleteProcThreadAttributeList(siEx.lpAttributeList);
134 free(siEx.lpAttributeList);
135
136 if (!success) {
137 spdlog::error("Failed to create process: {}", GetLastError());
138 ClosePseudoConsole(m_hPC);
139 CloseHandle(m_hPipeIn);
140 CloseHandle(m_hPipeOut);
141 return false;
142 }
143
144 m_hProcess = pi.hProcess;
145 m_hThread = pi.hThread;
146
147 spdlog::info("ConPTY session started: PID={}, Command=\"{}\"",
148 pi.dwProcessId, std::string(commandline.begin(), commandline.end()));
149
150 // Start I/O threads
151 m_running = true;
152 m_outputThread = std::thread(&ConPtySession::ReadOutputThread, this);
153 m_monitorThread = std::thread(&ConPtySession::ProcessMonitorThread, this);
154
155 return true;

158void ConPtySession::Stop() {
159 if (!m_running) {
160 return;
161 }
162
163 spdlog::info("Stopping ConPTY session");
164 m_running = false;
165
166 // First close the pseudoconsole - this signals the child process to exit
167 // and causes ReadFile to return cleanly
168 if (m_hPC != INVALID_HANDLE_VALUE) {
169 ClosePseudoConsole(m_hPC);
170 m_hPC = INVALID_HANDLE_VALUE;
171 }
172
173 // Now close pipe handles to unblock any remaining reads
174 if (m_hPipeOut != INVALID_HANDLE_VALUE) {
175 CancelIoEx(m_hPipeOut, nullptr);
176 CloseHandle(m_hPipeOut);
177 m_hPipeOut = INVALID_HANDLE_VALUE;
178 }
179
180 if (m_hPipeIn != INVALID_HANDLE_VALUE) {
181 CloseHandle(m_hPipeIn);
182 m_hPipeIn = INVALID_HANDLE_VALUE;
183 }
184
185 // Wait for threads to finish
186 if (m_outputThread.joinable()) {
187 m_outputThread.join();
188 }
189
190 if (m_monitorThread.joinable()) {
191 m_monitorThread.join();
192 }
193
194 // Terminate process if still running
195 if (m_hProcess != INVALID_HANDLE_VALUE) {
196 TerminateProcess(m_hProcess, 0);
197 CloseHandle(m_hProcess);
198 m_hProcess = INVALID_HANDLE_VALUE;
199 }
200
201 if (m_hThread != INVALID_HANDLE_VALUE) {
202 CloseHandle(m_hThread);
203 m_hThread = INVALID_HANDLE_VALUE;
204 }
205}

255void ConPtySession::ReadOutputThread() {
256 spdlog::debug("Output read thread started");
257
258 const size_t bufferSize = 4096;
259 std::vector<char> buffer(bufferSize);
260
261 while (m_running) {
262 DWORD bytesRead = 0;
263 BOOL success = ReadFile(m_hPipeOut, buffer.data(), static_cast<DWORD>(bufferSize),
264 &bytesRead, nullptr);
265
266 if (!success || bytesRead == 0) {
267 if (m_running) {
268 DWORD error = GetLastError();
269 if (error != ERROR_BROKEN_PIPE) {
270 spdlog::error("ReadFile failed: {}", error);
271 }
272 }
273 break;
274 }
275
276 // Call output callback if set
277 if (m_outputCallback && bytesRead > 0) {
278 m_outputCallback(buffer.data(), bytesRead);
279 }
280 }
281
282 spdlog::debug("Output read thread stopped");
283}

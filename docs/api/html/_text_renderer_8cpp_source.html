5// Convert UTF-8 string to UTF-32
6std::u32string Utf8ToUtf32(const std::string& utf8) {
7 std::u32string result;
8 size_t i = 0;
9 while (i < utf8.size()) {
10 char32_t codepoint = 0;
11 unsigned char c = utf8[i];
12 if ((c & 0x80) == 0) {
13 codepoint = c;
14 i += 1;
15 } else if ((c & 0xE0) == 0xC0) {
16 codepoint = (c & 0x1F) << 6;
17 if (i + 1 < utf8.size()) codepoint |= (utf8[i + 1] & 0x3F);
18 i += 2;
19 } else if ((c & 0xF0) == 0xE0) {
20 codepoint = (c & 0x0F) << 12;
21 if (i + 1 < utf8.size()) codepoint |= (utf8[i + 1] & 0x3F) << 6;
22 if (i + 2 < utf8.size()) codepoint |= (utf8[i + 2] & 0x3F);
23 i += 3;
24 } else if ((c & 0xF8) == 0xF0) {
25 codepoint = (c & 0x07) << 18;
26 if (i + 1 < utf8.size()) codepoint |= (utf8[i + 1] & 0x3F) << 12;
27 if (i + 2 < utf8.size()) codepoint |= (utf8[i + 2] & 0x3F) << 6;
28 if (i + 3 < utf8.size()) codepoint |= (utf8[i + 3] & 0x3F);
29 i += 4;
30 } else {
31 i += 1; // Skip invalid byte
32 }
33 if (codepoint > 0) result += codepoint;
34 }
35 return result;
36}

141void TextRenderer::RenderText(const std::u32string& text, float x, float y, const XMFLOAT4& color) {
142 if (!m_atlas) {
143 spdlog::error("GlyphAtlas not set");
144 return;
145 }
146
147 float currentX = x;
148 float currentY = y;
149
150 for (char32_t ch : text) {
151 // Handle newline
152 if (ch == U'\n') {
153 currentX = x;
154 currentY += m_atlas->GetLineHeight();
155 continue;
156 }
157
158 // Get glyph from atlas
159 const GlyphInfo* glyph = m_atlas->GetGlyph(ch);
160 if (!glyph) {
161 // Use space character as fallback
162 glyph = m_atlas->GetGlyph(U' ');
163 if (!glyph) continue;
164 }
165
166 // Create instance data
167 GlyphInstance instance;
168
169 // Position (account for glyph bearing)
170 instance.position.x = currentX + glyph->bearingX;
171 instance.position.y = currentY - glyph->bearingY + m_atlas->GetFontSize();
172
173 // Size
174 instance.size.x = static_cast<float>(glyph->width);
175 instance.size.y = static_cast<float>(glyph->height);
176
177 // UV coordinates
178 instance.uvMin.x = glyph->u0;
179 instance.uvMin.y = glyph->v0;
180 instance.uvMax.x = glyph->u1;
181 instance.uvMax.y = glyph->v1;
182
183 // Color
184 instance.color = color;
185
186 // Add to instance list
187 if (m_instances.size() < MAX_INSTANCES) {
188 m_instances.push_back(instance);
189 } else {
190 spdlog::warn("Max instances reached, skipping glyphs");
191 break;
192 }
193
194 // Advance cursor
195 currentX += glyph->advance;
196 }

316void TextRenderer::Render(ID3D12GraphicsCommandList* commandList) {
317 static int renderCallCount = 0;
318
319 if (m_instances.empty()) {
320 if (renderCallCount < 5) {
321 spdlog::info("TextRenderer::Render - no instances to render");
322 renderCallCount++;
323 }
324 return;
325 }
326
327 if (!m_rootSignature || !m_pipelineState || !m_atlas) {
328 spdlog::error("TextRenderer::Render - missing resources (rootSig:{}, PSO:{}, atlas:{})",
329 (void*)m_rootSignature, (void*)m_pipelineState, (void*)m_atlas);
330 return;
331 }
332
333 if (renderCallCount < 5) {
334 spdlog::info("TextRenderer::Render - rendering {} instances", m_instances.size());
335 renderCallCount++;
336 }
337
338 // Upload atlas to GPU if it has new glyphs
339 m_atlas->UploadAtlasIfDirty();
340
341 // Update instance buffer with current data
342 UpdateInstanceBuffer();
343
344 // Copy from upload buffer to default buffer
345 commandList->CopyBufferRegion(
346 m_instanceBuffer.Get(),
347 0,
348 m_instanceUploadBuffer.Get(),
349 0,
350 sizeof(GlyphInstance) * m_instances.size()
351 );
352
353 // Transition instance buffer to vertex buffer state
354 D3D12_RESOURCE_BARRIER barrier = {};
355 barrier.Type = D3D12_RESOURCE_BARRIER_TYPE_TRANSITION;
356 barrier.Transition.pResource = m_instanceBuffer.Get();
357 barrier.Transition.StateBefore = D3D12_RESOURCE_STATE_COPY_DEST;
358 barrier.Transition.StateAfter = D3D12_RESOURCE_STATE_VERTEX_AND_CONSTANT_BUFFER;
359 barrier.Transition.Subresource = D3D12_RESOURCE_BARRIER_ALL_SUBRESOURCES;
360 commandList->ResourceBarrier(1, &barrier);
361
362 // Set pipeline state
363 commandList->SetPipelineState(m_pipelineState);
364 commandList->SetGraphicsRootSignature(m_rootSignature);
365
366 // Set descriptor heap for atlas texture
367 ID3D12DescriptorHeap* heaps[] = { m_atlas->GetSRVHeap() };
368 commandList->SetDescriptorHeaps(1, heaps);
369
370 // Set root parameters (screen dimensions, atlas texture, etc.)
371 struct ScreenConstants {
372 float screenWidth;
373 float screenHeight;
374 float padding[2];
375 };
376
377 ScreenConstants constants = {
378 static_cast<float>(m_screenWidth),
379 static_cast<float>(m_screenHeight),
380 {0, 0}
381 };
382
383 commandList->SetGraphicsRoot32BitConstants(0, sizeof(ScreenConstants) / 4, &constants, 0);
384 commandList->SetGraphicsRootDescriptorTable(1, m_atlas->GetAtlasSRV());
385
386 // Set instance buffer
387 commandList->IASetVertexBuffers(0, 1, &m_instanceBufferView);
388
389 // Set primitive topology
390 commandList->IASetPrimitiveTopology(D3D_PRIMITIVE_TOPOLOGY_TRIANGLESTRIP);
391
392 // Draw instanced quads (4 vertices per quad, instanced by glyph count)
393 commandList->DrawInstanced(4, static_cast<UINT>(m_instances.size()), 0, 0);
394
395 // Transition back to copy dest for next frame
396 barrier.Transition.StateBefore = D3D12_RESOURCE_STATE_VERTEX_AND_CONSTANT_BUFFER;
397 barrier.Transition.StateAfter = D3D12_RESOURCE_STATE_COPY_DEST;
398 commandList->ResourceBarrier(1, &barrier);
399}

19void VTStateMachine::ProcessInput(const char* data, size_t size) {
20 for (size_t i = 0; i < size; ++i) {
21 uint8_t byte = static_cast<uint8_t>(data[i]);
22
23 // If we're in an escape sequence, process bytes directly
24 if (m_state != State::Ground) {
25 ProcessCharacter(data[i]);
26 continue;
27 }
28
29 // UTF-8 decoding for Ground state
30 if (m_utf8BytesNeeded > 0) {
31 // Expecting continuation byte (10xxxxxx)
32 if ((byte & 0xC0) == 0x80) {
33 m_utf8Buffer[m_utf8BytesReceived++] = byte;
34 if (m_utf8BytesReceived == m_utf8BytesNeeded) {
35 // Decode complete UTF-8 sequence
36 char32_t codepoint = 0;
37 if (m_utf8BytesNeeded == 2) {
38 codepoint = ((m_utf8Buffer[0] & 0x1F) << 6) |
39 (m_utf8Buffer[1] & 0x3F);
40 } else if (m_utf8BytesNeeded == 3) {
41 codepoint = ((m_utf8Buffer[0] & 0x0F) << 12) |
42 ((m_utf8Buffer[1] & 0x3F) << 6) |
43 (m_utf8Buffer[2] & 0x3F);
44 } else if (m_utf8BytesNeeded == 4) {
45 codepoint = ((m_utf8Buffer[0] & 0x07) << 18) |
46 ((m_utf8Buffer[1] & 0x3F) << 12) |
47 ((m_utf8Buffer[2] & 0x3F) << 6) |
48 (m_utf8Buffer[3] & 0x3F);
49 }
50 m_utf8BytesNeeded = 0;
51 m_utf8BytesReceived = 0;
52 // Write the decoded codepoint
53 m_screenBuffer->WriteChar(codepoint);
54 }
55 } else {
56 // Invalid continuation byte - reset and process as new byte
57 m_utf8BytesNeeded = 0;
58 m_utf8BytesReceived = 0;
59 // Fall through to process this byte
60 }
61 }
62
63 if (m_utf8BytesNeeded == 0) {
64 // Check for UTF-8 lead byte
65 if ((byte & 0x80) == 0) {
66 // ASCII (0xxxxxxx) - process directly
67 ProcessCharacter(data[i]);
68 } else if ((byte & 0xE0) == 0xC0) {
69 // 2-byte sequence (110xxxxx)
70 m_utf8Buffer[0] = byte;
71 m_utf8BytesNeeded = 2;
72 m_utf8BytesReceived = 1;
73 } else if ((byte & 0xF0) == 0xE0) {
74 // 3-byte sequence (1110xxxx)
75 m_utf8Buffer[0] = byte;
76 m_utf8BytesNeeded = 3;
77 m_utf8BytesReceived = 1;
78 } else if ((byte & 0xF8) == 0xF0) {
79 // 4-byte sequence (11110xxx)
80 m_utf8Buffer[0] = byte;
81 m_utf8BytesNeeded = 4;
82 m_utf8BytesReceived = 1;
83 } else {
84 // Invalid UTF-8 lead byte - skip
85 }
86 }
87 }

90void VTStateMachine::ProcessCharacter(char ch) {
91 switch (m_state) {
92 case State::Ground:
93 if (ch == '\x1b') { // ESC
94 m_state = State::Escape;
95 ResetState();
96 } else {
97 ExecuteCharacter(ch);
98 }
99 break;
100
101 case State::Escape:
102 HandleEscapeSequence(ch);
103 break;
104
105 case State::CSI_Entry:
106 // Just entered CSI - private mode indicators are only valid here
107 if (ch == '\x1b') {
108 // ESC during CSI - abort and start new escape sequence
109 m_state = State::Escape;
110 ResetState();
111 } else if (ch < 0x20 && ch != '\x1b') {
112 // C0 control characters - execute them (CR, LF, etc.)
113 ExecuteCharacter(ch);
114 } else if (ch >= '0' && ch <= '9') {
115 m_paramBuffer += ch;
116 m_state = State::CSI_Param;
117 } else if (ch == ';') {
118 // Empty first parameter
119 m_params.push_back(0);
120 m_state = State::CSI_Param;
121 } else if (ch == '?' || ch == '>' || ch == '!') {
122 // Private mode indicator - ONLY valid at CSI_Entry (immediately after ESC[)
123 m_intermediateChar = ch;
124 m_state = State::CSI_Param;
125 } else if (ch >= 0x20 && ch <= 0x2F) { // Intermediate byte
126 m_intermediateChar = ch;
127 m_state = State::CSI_Intermediate;
128 } else if (ch >= 0x40 && ch <= 0x7E) { // Final byte
129 m_finalChar = ch;
130 HandleCSI();
131 m_state = State::Ground;
132 } else {
133 // Invalid character - abort sequence and return to ground
134 m_state = State::Ground;
135 }
136 break;
137
138 case State::CSI_Param:
139 // Parsing CSI parameters - private mode indicators are NOT valid here
140 if (ch == '\x1b') {
141 // ESC during CSI - abort and start new escape sequence
142 m_state = State::Escape;
143 ResetState();
144 } else if (ch < 0x20 && ch != '\x1b') {
145 // C0 control characters - execute them (CR, LF, etc.)
146 ExecuteCharacter(ch);
147 } else if (ch >= '0' && ch <= '9') {
148 m_paramBuffer += ch;
149 } else if (ch == ';') {
150 // Parse current parameter
151 if (!m_paramBuffer.empty()) {
152 m_params.push_back(std::stoi(m_paramBuffer));
153 m_paramBuffer.clear();
154 } else {
155 m_params.push_back(0);
156 }
157 } else if (ch >= 0x20 && ch <= 0x2F) { // Intermediate byte
158 // Parse pending parameter first
159 if (!m_paramBuffer.empty()) {
160 m_params.push_back(std::stoi(m_paramBuffer));
161 m_paramBuffer.clear();
162 }
163 m_intermediateChar = ch;
164 m_state = State::CSI_Intermediate;
165 } else if (ch >= 0x40 && ch <= 0x7E) { // Final byte
166 // Parse last parameter if any
167 if (!m_paramBuffer.empty()) {
168 m_params.push_back(std::stoi(m_paramBuffer));
169 }
170 m_finalChar = ch;
171 HandleCSI();
172 m_state = State::Ground;
173 } else {
174 // Invalid character (including ?, >, ! after parameters) - abort
175 m_state = State::Ground;
176 }
177 break;
178
179 case State::CSI_Intermediate:
180 // Handle control characters during CSI intermediate
181 if (ch == '\x1b') {
182 // ESC - abort and start new escape sequence
183 m_state = State::Escape;
184 ResetState();
185 } else if (ch < 0x20) {
186 // C0 control characters - execute them
187 ExecuteCharacter(ch);
188 } else if (ch >= 0x40 && ch <= 0x7E) {
189 m_finalChar = ch;
190 HandleCSI();
191 m_state = State::Ground;
192 } else if (ch >= 0x80) {
193 // Invalid - abort sequence
194 m_state = State::Ground;
195 }
196 break;
197
198 case State::OSC_String:
199 // OSC sequences end with BEL (0x07) or ST (ESC \‍)
200 if (ch == '\x07') { // BEL - end of OSC
201 HandleOSC();
202 m_state = State::Ground;
203 } else if (ch == '\x1b') {
204 // Might be ST (String Terminator) - ESC \
205 // For simplicity, just end the OSC on any ESC
206 HandleOSC();
207 m_state = State::Escape;
208 ResetState();
209 } else if (ch == '\x9c') {
210 // C1 ST (String Terminator)
211 HandleOSC();
212 m_state = State::Ground;
213 } else {
214 // Accumulate OSC content
215 m_oscBuffer += ch;
216 // Sanity limit to prevent unbounded memory growth
217 if (m_oscBuffer.size() > 4096) {
218 m_oscBuffer.clear();
219 m_state = State::Ground;
220 }
221 }
222 break;
223
224 default:
225 m_state = State::Ground;
226 break;
227 }

230void VTStateMachine::HandleEscapeSequence(char ch) {
231 switch (ch) {
232 case '[': // CSI
233 m_state = State::CSI_Entry;
234 break;
235
236 case ']': // OSC (Operating System Command)
237 m_state = State::OSC_String;
238 m_oscBuffer.clear();
239 break;
240
241 case 'M': { // Reverse Index (RI) - move up, scroll if at top
242 int y = m_screenBuffer->GetCursorY();
243 int scrollTop = m_screenBuffer->GetScrollRegionTop();
244 if (y <= scrollTop) {
245 // At top of scroll region - scroll content down
246 m_screenBuffer->ScrollRegionDown(1);
247 } else {
248 // Not at top - just move cursor up
249 m_screenBuffer->SetCursorPos(m_screenBuffer->GetCursorX(), y - 1);
250 }
251 m_state = State::Ground;
252 break;
253 }
254
255 case 'D': { // Index (IND) - move down, scroll if at bottom
256 int y = m_screenBuffer->GetCursorY();
257 int scrollBottom = m_screenBuffer->GetScrollRegionBottom();
258 if (y >= scrollBottom) {
259 // At bottom of scroll region - scroll content up
260 m_screenBuffer->ScrollRegionUp(1);
261 } else {
262 // Not at bottom - just move cursor down
263 m_screenBuffer->SetCursorPos(m_screenBuffer->GetCursorX(), y + 1);
264 }
265 m_state = State::Ground;
266 break;
267 }
268
269 case 'E': // Next Line
270 m_screenBuffer->WriteChar('\n');
271 m_state = State::Ground;
272 break;
273
274 case 'c': // Reset
275 m_screenBuffer->Clear();
276 m_screenBuffer->SetCursorPos(0, 0);
277 m_state = State::Ground;
278 break;
279
280 case '7': // DECSC - Save cursor
281 m_savedCursor.x = m_screenBuffer->GetCursorX();
282 m_savedCursor.y = m_screenBuffer->GetCursorY();
283 m_savedCursor.attr = m_screenBuffer->GetCurrentAttributes();
284 m_savedCursor.originMode = m_originMode;
285 m_savedCursor.autoWrap = m_autoWrap;
286 m_savedCursor.valid = true;
287 m_state = State::Ground;
288 break;
289
290 case '8': // DECRC - Restore cursor
291 if (m_savedCursor.valid) {
292 m_screenBuffer->SetCursorPos(m_savedCursor.x, m_savedCursor.y);
293 m_screenBuffer->SetCurrentAttributes(m_savedCursor.attr);
294 m_originMode = m_savedCursor.originMode;
295 m_autoWrap = m_savedCursor.autoWrap;
296 }
297 m_state = State::Ground;
298 break;
299
300 case 'H': // HTS - Horizontal Tab Set
301 m_screenBuffer->SetTabStop(m_screenBuffer->GetCursorX());
302 m_state = State::Ground;
303 break;
304
305 case '=': // DECKPAM - Keypad Application Mode
306 m_keypadApplicationMode = true;
307 spdlog::debug("DECKPAM: Keypad application mode enabled");
308 m_state = State::Ground;
309 break;
310
311 case '>': // DECKPNM - Keypad Numeric Mode
312 m_keypadApplicationMode = false;
313 spdlog::debug("DECKPNM: Keypad numeric mode enabled");
314 m_state = State::Ground;
315 break;
316
317 default:
318 // Unknown escape sequence, ignore
319 m_state = State::Ground;
320 break;
321 }
322}

590void VTStateMachine::HandleSGR() {
591 // If no parameters, default to 0 (reset)
592 if (m_params.empty()) {
593 m_params.push_back(0);
594 }
595
596 // Debug log SGR parameters
597 std::string paramStr;
598 for (size_t i = 0; i < m_params.size(); ++i) {
599 if (i > 0) paramStr += ";";
600 paramStr += std::to_string(m_params[i]);
601 }
602 spdlog::debug("SGR sequence: ESC[{}m", paramStr);
603
604 CellAttributes attr = m_screenBuffer->GetCurrentAttributes();
605
606 for (size_t i = 0; i < m_params.size(); ++i) {
607 int param = m_params[i];
608 switch (param) {
609 case 0: // Reset
610 attr.foreground = 7; // White
611 attr.background = 0; // Black
612 attr.flags = 0;
613 attr.flags2 = 0;
614 attr.underlineStyle = CellAttributes::UnderlineStyle::None;
615 break;
616
617 case 1: // Bold
618 attr.flags |= CellAttributes::FLAG_BOLD;
619 break;
620
621 case 2: // Dim/Faint
622 attr.flags |= CellAttributes::FLAG_DIM;
623 break;
624
625 case 3: // Italic
626 attr.flags |= CellAttributes::FLAG_ITALIC;
627 break;
628
629 case 4: // Underline (single)
630 attr.underlineStyle = CellAttributes::UnderlineStyle::Single;
631 attr.flags |= CellAttributes::FLAG_UNDERLINE;
632 break;
633
634 case 5: // Slow Blink
635 case 6: // Rapid Blink (treat same as slow blink)
636 attr.flags2 |= CellAttributes::FLAG2_BLINK;
637 break;
638
639 case 7: // Inverse
640 attr.flags |= CellAttributes::FLAG_INVERSE;
641 break;
642
643 case 8: // Hidden/Invisible
644 attr.flags2 |= CellAttributes::FLAG2_HIDDEN;
645 break;
646
647 case 9: // Strikethrough
648 attr.flags |= CellAttributes::FLAG_STRIKETHROUGH;
649 break;
650
651 case 21: // Double underline
652 attr.underlineStyle = CellAttributes::UnderlineStyle::Double;
653 attr.flags |= CellAttributes::FLAG_UNDERLINE;
654 break;
655
656 case 22: // Not bold/dim - clears both
657 attr.flags &= ~(CellAttributes::FLAG_BOLD | CellAttributes::FLAG_DIM);
658 break;
659
660 case 23: // Not italic
661 attr.flags &= ~CellAttributes::FLAG_ITALIC;
662 break;
663
664 case 24: // Not underline
665 attr.underlineStyle = CellAttributes::UnderlineStyle::None;
666 attr.flags &= ~CellAttributes::FLAG_UNDERLINE;
667 break;
668
669 case 25: // Not blinking
670 attr.flags2 &= ~CellAttributes::FLAG2_BLINK;
671 break;
672
673 case 27: // Not inverse
674 attr.flags &= ~CellAttributes::FLAG_INVERSE;
675 break;
676
677 case 28: // Not hidden
678 attr.flags2 &= ~CellAttributes::FLAG2_HIDDEN;
679 break;
680
681 case 29: // Not strikethrough
682 attr.flags &= ~CellAttributes::FLAG_STRIKETHROUGH;
683 break;
684
685 // Foreground colors (30-37 for standard, 90-97 for bright)
686 case 30: case 31: case 32: case 33: case 34: case 35: case 36: case 37:
687 attr.foreground = param - 30;
688 break;
689
690 case 90: case 91: case 92: case 93: case 94: case 95: case 96: case 97:
691 attr.foreground = param - 90 + 8; // Bright colors are 8-15
692 break;
693
694 // Extended foreground color: 38;5;N (256-color) or 38;2;R;G;B (true color)
695 case 38:
696 if (i + 1 < m_params.size()) {
697 if (m_params[i + 1] == 5 && i + 2 < m_params.size()) {
698 // 256-color mode: 38;5;N
699 int colorIndex = m_params[i + 2];
700 // Map 256-color to 16-color for now (first 16 are the same)
701 if (colorIndex < 16) {
702 attr.foreground = colorIndex;
703 } else if (colorIndex >= 232) {
704 // Grayscale ramp (232-255) -> map to closest standard color
705 int gray = (colorIndex - 232) * 255 / 23;
706 if (gray < 64) attr.foreground = 0; // Black
707 else if (gray < 192) attr.foreground = 8; // Gray
708 else attr.foreground = 7; // White
709 } else {
710 // Color cube (16-231) -> map to closest 16-color
711 // Color cube is 6x6x6: index = 16 + 36*r + 6*g + b where r,g,b in [0,5]
712 int idx = colorIndex - 16;
713 int r = idx / 36;
714 int g = (idx % 36) / 6;
715 int b = idx % 6;
716 // Map to standard colors based on which channel is highest
717 if (r > g && r > b) attr.foreground = (r > 3) ? 9 : 1; // Red
718 else if (g > r && g > b) attr.foreground = (g > 3) ? 10 : 2; // Green
719 else if (b > r && b > g) attr.foreground = (b > 3) ? 12 : 4; // Blue
720 else if (r == g && r > b) attr.foreground = (r > 3) ? 11 : 3; // Yellow
721 else if (r == b && r > g) attr.foreground = (r > 3) ? 13 : 5; // Magenta
722 else if (g == b && g > r) attr.foreground = (g > 3) ? 14 : 6; // Cyan
723 else attr.foreground = (r > 2) ? 15 : 7; // White/Gray
724 }
725 i += 2; // Skip the 5 and N parameters
726 } else if (m_params[i + 1] == 2 && i + 4 < m_params.size()) {
727 // True color mode: 38;2;R;G;B - store actual RGB
728 attr.SetForegroundRGB(
729 static_cast<uint8_t>(m_params[i + 2]),
730 static_cast<uint8_t>(m_params[i + 3]),
731 static_cast<uint8_t>(m_params[i + 4]));
732 i += 4; // Skip the 2, R, G, B parameters
733 }
734 }
735 break;
736
737 // Background colors (40-47 for standard, 100-107 for bright)
738 case 40: case 41: case 42: case 43: case 44: case 45: case 46: case 47:
739 attr.background = param - 40;
740 break;
741
742 case 100: case 101: case 102: case 103: case 104: case 105: case 106: case 107:
743 attr.background = param - 100 + 8;
744 break;
745
746 // Extended background color: 48;5;N (256-color) or 48;2;R;G;B (true color)
747 case 48:
748 if (i + 1 < m_params.size()) {
749 if (m_params[i + 1] == 5 && i + 2 < m_params.size()) {
750 // 256-color mode: 48;5;N
751 int colorIndex = m_params[i + 2];
752 if (colorIndex < 16) {
753 attr.background = colorIndex;
754 } else if (colorIndex >= 232) {
755 int gray = (colorIndex - 232) * 255 / 23;
756 if (gray < 64) attr.background = 0;
757 else if (gray < 192) attr.background = 8;
758 else attr.background = 7;
759 } else {
760 int idx = colorIndex - 16;
761 int r = idx / 36;
762 int g = (idx % 36) / 6;
763 int b = idx % 6;
764 if (r > g && r > b) attr.background = (r > 3) ? 9 : 1;
765 else if (g > r && g > b) attr.background = (g > 3) ? 10 : 2;
766 else if (b > r && b > g) attr.background = (b > 3) ? 12 : 4;
767 else if (r == g && r > b) attr.background = (r > 3) ? 11 : 3;
768 else if (r == b && r > g) attr.background = (r > 3) ? 13 : 5;
769 else if (g == b && g > r) attr.background = (g > 3) ? 14 : 6;
770 else attr.background = (r > 2) ? 15 : 7;
771 }
772 i += 2;
773 } else if (m_params[i + 1] == 2 && i + 4 < m_params.size()) {
774 // True color mode: 48;2;R;G;B - store actual RGB
775 attr.SetBackgroundRGB(
776 static_cast<uint8_t>(m_params[i + 2]),
777 static_cast<uint8_t>(m_params[i + 3]),
778 static_cast<uint8_t>(m_params[i + 4]));
779 i += 4;
780 }
781 }
782 break;
783
784 case 39: // Default foreground
785 attr.foreground = 7;
786 break;
787
788 case 49: // Default background
789 attr.background = 0;
790 break;
791 }
792 }
793
794 m_screenBuffer->SetCurrentAttributes(attr);
795}

906void VTStateMachine::HandleOSC() {
907 // OSC sequences: ESC ] Ps ; Pt BEL
908 // Ps = 0: Change icon name and window title
909 // Ps = 1: Change icon name
910 // Ps = 2: Change window title
911 // Ps = 133: Shell integration (FinalTerm/iTerm2)
912
913 if (m_oscBuffer.empty()) {
914 return;
915 }
916
917 // Parse the OSC type (number before first semicolon)
918 size_t semicolon = m_oscBuffer.find(';');
919 if (semicolon != std::string::npos) {
920 std::string type = m_oscBuffer.substr(0, semicolon);
921 std::string value = m_oscBuffer.substr(semicolon + 1);
922
923 spdlog::debug("OSC sequence: type={}, value=\"{}\"", type, value);
924
925 // OSC 133 - Shell integration
926 if (type == "133") {
927 HandleOSC133(value);
928 }
929 // OSC 10 - Set/Query foreground color
930 else if (type == "10") {
931 HandleOSC10(value);
932 }
933 // OSC 11 - Set/Query background color
934 else if (type == "11") {
935 HandleOSC11(value);
936 }
937 // OSC 8 - Hyperlink
938 else if (type == "8") {
939 HandleOSC8(value);
940 }
941 // OSC 52 - Clipboard access
942 else if (type == "52") {
943 HandleOSC52(value);
944 }
945 // OSC 4 - Set/Query palette color
946 else if (type == "4") {
947 HandleOSC4(value);
948 }
949 // OSC 0, 1, 2 are window title - we could handle these if desired
950 }
951
952 m_oscBuffer.clear();
953}

955void VTStateMachine::HandleOSC133(const std::string& param) {
956 // OSC 133 shell integration:
957 // A = Prompt start (before prompt is displayed)
958 // B = Prompt end / Input start (after prompt, before user types)
959 // C = Command start (user pressed Enter)
960 // D = Command finished (optionally with exit code: D;N)
961
962 if (param.empty()) return;
963
964 char marker = param[0];
965 switch (marker) {
966 case 'A':
967 m_screenBuffer->MarkPromptStart();
968 break;
969 case 'B':
970 m_screenBuffer->MarkInputStart();
971 break;
972 case 'C':
973 m_screenBuffer->MarkCommandStart();
974 break;
975 case 'D': {
976 int exitCode = -1;
977 // Check for exit code: D;N
978 if (param.length() > 2 && param[1] == ';') {
979 try {
980 exitCode = std::stoi(param.substr(2));
981 } catch (...) {
982 exitCode = -1;
983 }
984 }
985 m_screenBuffer->MarkCommandEnd(exitCode);
986 break;
987 }
988 default:
989 spdlog::debug("Unknown OSC 133 marker: {}", marker);
990 break;
991 }
992}

994void VTStateMachine::HandleOSC8(const std::string& param) {
995 // OSC 8 - Hyperlink
996 // Format: OSC 8 ; params ; uri ST
997 // params can include id=xxx for link grouping
998 // Empty uri ends the hyperlink
999
1000 // param format: "params;uri" where params can be empty
1001 size_t semicolon = param.find(';');
1002 if (semicolon == std::string::npos) {
1003 // Invalid format - just ignore
1004 return;
1005 }
1006
1007 std::string params = param.substr(0, semicolon);
1008 std::string uri = param.substr(semicolon + 1);
1009
1010 // Parse optional ID from params
1011 std::string linkId;
1012 if (!params.empty()) {
1013 // params are key=value pairs separated by ':'
1014 size_t pos = 0;
1015 while (pos < params.length()) {
1016 size_t colonPos = params.find(':', pos);
1017 std::string kvPair = (colonPos == std::string::npos)
1018 ? params.substr(pos)
1019 : params.substr(pos, colonPos - pos);
1020
1021 size_t eqPos = kvPair.find('=');
1022 if (eqPos != std::string::npos) {
1023 std::string key = kvPair.substr(0, eqPos);
1024 std::string value = kvPair.substr(eqPos + 1);
1025 if (key == "id") {
1026 linkId = value;
1027 }
1028 }
1029
1030 if (colonPos == std::string::npos) break;
1031 pos = colonPos + 1;
1032 }
1033 }
1034
1035 if (uri.empty()) {
1036 // End hyperlink
1037 m_screenBuffer->ClearCurrentHyperlink();
1038 spdlog::debug("OSC 8: End hyperlink");
1039 } else {
1040 // Start hyperlink
1041 m_screenBuffer->AddHyperlink(uri, linkId);
1042 spdlog::debug("OSC 8: Start hyperlink uri=\"{}\" id=\"{}\"", uri, linkId);
1043 }
1044}

1046bool VTStateMachine::ParseOSCColor(const std::string& colorStr, uint8_t& r, uint8_t& g, uint8_t& b) {
1047 // Supported formats:
1048 // rgb:RR/GG/BB - X11 format with 2 hex digits per component
1049 // rgb:RRRR/GGGG/BBBB - X11 format with 4 hex digits per component
1050 // #RRGGBB - CSS-style hex format
1051
1052 if (colorStr.empty()) return false;
1053
1054 if (colorStr[0] == '#' && colorStr.length() == 7) {
1055 // #RRGGBB format
1056 try {
1057 r = static_cast<uint8_t>(std::stoi(colorStr.substr(1, 2), nullptr, 16));
1058 g = static_cast<uint8_t>(std::stoi(colorStr.substr(3, 2), nullptr, 16));
1059 b = static_cast<uint8_t>(std::stoi(colorStr.substr(5, 2), nullptr, 16));
1060 return true;
1061 } catch (...) {
1062 return false;
1063 }
1064 }
1065
1066 if (colorStr.substr(0, 4) == "rgb:") {
1067 std::string rgb = colorStr.substr(4);
1068 size_t slash1 = rgb.find('/');
1069 size_t slash2 = rgb.rfind('/');
1070 if (slash1 != std::string::npos && slash2 != std::string::npos && slash1 != slash2) {
1071 try {
1072 std::string rStr = rgb.substr(0, slash1);
1073 std::string gStr = rgb.substr(slash1 + 1, slash2 - slash1 - 1);
1074 std::string bStr = rgb.substr(slash2 + 1);
1075
1076 // Handle both 2-digit and 4-digit formats
1077 int rVal = std::stoi(rStr, nullptr, 16);
1078 int gVal = std::stoi(gStr, nullptr, 16);
1079 int bVal = std::stoi(bStr, nullptr, 16);
1080
1081 // If 4-digit format, scale down to 8-bit
1082 if (rStr.length() == 4) rVal = rVal >> 8;
1083 if (gStr.length() == 4) gVal = gVal >> 8;
1084 if (bStr.length() == 4) bVal = bVal >> 8;
1085
1086 r = static_cast<uint8_t>(rVal);
1087 g = static_cast<uint8_t>(gVal);
1088 b = static_cast<uint8_t>(bVal);
1089 return true;
1090 } catch (...) {
1091 return false;
1092 }
1093 }
1094 }
1095
1096 return false;
1097}

1147// Base64 encoding table
1148static const char base64_chars[] =
1149 "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
1150
1151static std::string Base64Encode(const std::string& input) {
1152 std::string output;
1153 output.reserve(((input.size() + 2) / 3) * 4);
1154
1155 for (size_t i = 0; i < input.size(); i += 3) {
1156 uint32_t n = static_cast<uint8_t>(input[i]) << 16;
1157 if (i + 1 < input.size()) n |= static_cast<uint8_t>(input[i + 1]) << 8;
1158 if (i + 2 < input.size()) n |= static_cast<uint8_t>(input[i + 2]);
1159
1160 output.push_back(base64_chars[(n >> 18) & 0x3F]);
1161 output.push_back(base64_chars[(n >> 12) & 0x3F]);
1162 output.push_back((i + 1 < input.size()) ? base64_chars[(n >> 6) & 0x3F] : '=');
1163 output.push_back((i + 2 < input.size()) ? base64_chars[n & 0x3F] : '=');
1164 }
1165 return output;
1166}
1167
1168static std::string Base64Decode(const std::string& input) {
1169 std::string output;
1170 std::vector<int> T(256, -1);
1171 for (int i = 0; i < 64; i++) T[static_cast<unsigned char>(base64_chars[i])] = i;
1172
1173 int val = 0, valb = -8;
1174 for (unsigned char c : input) {
1175 if (T[c] == -1) break;
1176 val = (val << 6) + T[c];
1177 valb += 6;
1178 if (valb >= 0) {
1179 output.push_back(char((val >> valb) & 0xFF));
1180 valb -= 8;
1181 }
1182 }
1183 return output;
1184}

1186void VTStateMachine::HandleOSC52(const std::string& param) {
1187 // OSC 52 - Clipboard access
1188 // Format: OSC 52 ; Pc ; Pd ST
1189 // Pc = clipboard selection ('c' for clipboard, 'p' for primary, 's' for select)
1190 // Pd = base64 encoded data, or '?' to query
1191
1192 size_t semicolon = param.find(';');
1193 if (semicolon == std::string::npos) {
1194 spdlog::debug("OSC 52: Invalid format (no semicolon)");
1195 return;
1196 }
1197
1198 std::string selection = param.substr(0, semicolon);
1199 std::string data = param.substr(semicolon + 1);
1200
1201 // We only support 'c' (system clipboard) on Windows
1202 // Ignore selection parameter and always use system clipboard
1203
1204 if (data == "?") {
1205 // Query clipboard
1206 if (m_clipboardReadCallback) {
1207 std::string clipboardContent = m_clipboardReadCallback();
1208 std::string encoded = Base64Encode(clipboardContent);
1209 std::string response = "]52;" + selection + ";" + encoded + "\x07";
1210 SendResponse(response);
1211 spdlog::debug("OSC 52: Query clipboard, {} bytes", clipboardContent.size());
1212 }
1213 } else if (!data.empty()) {
1214 // Set clipboard
1215 std::string decoded = Base64Decode(data);
1216 if (m_clipboardWriteCallback && !decoded.empty()) {
1217 m_clipboardWriteCallback(decoded);
1218 spdlog::debug("OSC 52: Set clipboard, {} bytes", decoded.size());
1219 }
1220 }
1221}

1223void VTStateMachine::HandleOSC4(const std::string& param) {
1224 // OSC 4 - Set/Query palette color
1225 // Format: OSC 4 ; index ; color ST
1226 // Query: OSC 4 ; index ; ? ST -> respond with rgb:RRRR/GGGG/BBBB
1227 // Multiple colors: OSC 4 ; index1 ; color1 ; index2 ; color2 ; ... ST
1228
1229 // Parse pairs of index;color
1230 size_t pos = 0;
1231 while (pos < param.length()) {
1232 // Find the index
1233 size_t semicolon1 = param.find(';', pos);
1234 if (semicolon1 == std::string::npos) break;
1235
1236 std::string indexStr = param.substr(pos, semicolon1 - pos);
1237 int index = 0;
1238 try {
1239 index = std::stoi(indexStr);
1240 } catch (...) {
1241 spdlog::debug("OSC 4: Invalid index: {}", indexStr);
1242 break;
1243 }
1244
1245 if (index < 0 || index >= 256) {
1246 spdlog::debug("OSC 4: Index out of range: {}", index);
1247 break;
1248 }
1249
1250 // Find the color (or next semicolon for another pair)
1251 size_t semicolon2 = param.find(';', semicolon1 + 1);
1252 std::string colorStr;
1253 if (semicolon2 == std::string::npos) {
1254 colorStr = param.substr(semicolon1 + 1);
1255 pos = param.length();
1256 } else {
1257 colorStr = param.substr(semicolon1 + 1, semicolon2 - semicolon1 - 1);
1258 pos = semicolon2 + 1;
1259 }
1260
1261 if (colorStr == "?") {
1262 // Query color - respond with current color
1263 const auto& paletteColor = m_screenBuffer->GetPaletteColor(index);
1264 // Format response as rgb:RRRR/GGGG/BBBB (16-bit components)
1265 char response[64];
1266 snprintf(response, sizeof(response), "]4;%d;rgb:%04x/%04x/%04x\x07",
1267 index,
1268 paletteColor.r * 257, // Scale 8-bit to 16-bit
1269 paletteColor.g * 257,
1270 paletteColor.b * 257);
1271 SendResponse(response);
1272 spdlog::debug("OSC 4: Query color {}", index);
1273 } else {
1274 // Set color
1275 uint8_t r, g, b;
1276 if (ParseOSCColor(colorStr, r, g, b)) {
1277 m_screenBuffer->SetPaletteColor(index, r, g, b);
1278 spdlog::debug("OSC 4: Set color {} to #{:02x}{:02x}{:02x}", index, r, g, b);
1279 }
1280 }
1281 }

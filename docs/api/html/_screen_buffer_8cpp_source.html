35void ScreenBuffer::Resize(int cols, int rows) {
36 if (cols == m_cols && rows == m_rows) {
37 return;
38 }
39
40 spdlog::info("Resizing screen buffer from {}x{} to {}x{}, altBuffer={}",
41 m_cols, m_rows, cols, rows, m_useAltBuffer ? "yes" : "no");
42
43 int oldCols = m_cols;
44 int oldRows = m_rows;
45
46 // For alternative buffer, just resize without reflow (TUI apps handle their own reflow)
47 if (m_useAltBuffer) {
48 std::vector<Cell> newBuffer(cols * rows);
49 int copyRows = std::min(oldRows, rows);
50 int copyCols = std::min(oldCols, cols);
51 for (int y = 0; y < copyRows; ++y) {
52 for (int x = 0; x < copyCols; ++x) {
53 newBuffer[y * cols + x] = m_buffer[y * oldCols + x];
54 }
55 }
56 m_buffer = std::move(newBuffer);
57 m_cols = cols;
58 m_rows = rows;
59 m_lineWrapped.resize(rows, false);
60 ClampCursor();
61
62 // Reset scroll region to full screen after resize
63 // This is critical: TUI apps set scroll regions, and after resize
64 // the old region bounds may be invalid, causing row 0 to be excluded
65 // from redraws. The app will re-establish its scroll region after
66 // receiving the resize notification.
67 m_scrollTop = 0;
68 m_scrollBottom = -1; // -1 means full screen (no custom region)
69
70 m_dirty = true;
71 return;
72 }
73
74 // Text reflow for main buffer
75 // Step 1: Extract all text as logical lines (combining soft-wrapped lines)
76 // Use m_lineWrapped flags to determine soft wraps - m_lineWrapped[y] = true means
77 // row y is a continuation of row y-1 (i.e., row y-1 soft-wrapped to row y)
78 std::vector<std::vector<Cell>> logicalLines;
79 std::vector<Cell> currentLine;
80
81 // Process visible buffer
82 for (int y = 0; y < oldRows; ++y) {
83 // Add cells from this row
84 for (int x = 0; x < oldCols; ++x) {
85 currentLine.push_back(m_buffer[y * oldCols + x]);
86 }
87
88 // Check if next row is a continuation of this row (soft wrap)
89 // m_lineWrapped[y+1] = true means row y soft-wrapped to row y+1
90 bool nextRowIsContinuation = false;
91 if (y + 1 < oldRows && y + 1 < static_cast<int>(m_lineWrapped.size())) {
92 nextRowIsContinuation = m_lineWrapped[y + 1];
93 }
94
95 // End the logical line if next row is NOT a continuation, or if this is the last line
96 if (!nextRowIsContinuation || y == oldRows - 1) {
97 // Trim trailing spaces from logical line
98 while (!currentLine.empty() && currentLine.back().ch == U' ') {
99 currentLine.pop_back();
100 }
101 logicalLines.push_back(std::move(currentLine));
102 currentLine.clear();
103 }
104 }
105
106 // Step 2: Re-wrap logical lines to new column width
107 std::vector<Cell> newBuffer(cols * rows);
108 std::vector<bool> newWrapped(rows, false);
109 int newRow = 0;
110 int newCol = 0;
111
112 for (const auto& logicalLine : logicalLines) {
113 if (newRow >= rows) break;
114
115 for (const Cell& cell : logicalLine) {
116 if (newRow >= rows) break;
117
118 newBuffer[newRow * cols + newCol] = cell;
119 newCol++;
120
121 // Wrap to next line if we've hit the column limit
122 if (newCol >= cols) {
123 newCol = 0;
124 newRow++;
125 if (newRow < rows) {
126 newWrapped[newRow] = true; // This is a soft wrap
127 }
128 }
129 }
130
131 // Move to next line after logical line
132 // Always advance to next row, even for empty lines
133 newRow++;
134 newCol = 0;
135 }
136
137 m_buffer = std::move(newBuffer);
138 m_lineWrapped = std::move(newWrapped);
139 m_cols = cols;
140 m_rows = rows;
141
142 // Resize the inactive alternative buffer (no reflow for alt buffer)
143 if (!m_altBuffer.empty()) {
144 std::vector<Cell> newAltBuffer(cols * rows);
145 int copyRows = std::min(oldRows, rows);
146 int copyCols = std::min(oldCols, cols);
147 for (int y = 0; y < copyRows; ++y) {
148 for (int x = 0; x < copyCols; ++x) {
149 newAltBuffer[y * cols + x] = m_altBuffer[y * oldCols + x];
150 }
151 }
152 m_altBuffer = std::move(newAltBuffer);
153 }
154
155 // Clamp cursor to new size
156 ClampCursor();
157
158 // Adjust scroll region to new size (don't reset - preserve TUI app's region)
159 // Only clamp the bottom if it exceeds new rows
160 if (m_scrollBottom >= rows) {
161 m_scrollBottom = rows - 1;
162 }
163 // Clamp top if needed
164 if (m_scrollTop >= rows) {
165 m_scrollTop = 0;
166 m_scrollBottom = -1; // Full reset if top was invalid
167 }
168
169 m_dirty = true;
170}

187void ScreenBuffer::WriteChar(char32_t ch, const CellAttributes& attr) {
188 // Handle control characters
189 switch (ch) {
190 case U'\n':
191 NewLine();
192 return;
193 case U'\r':
194 CarriageReturn();
195 return;
196 case U'\t':
197 Tab();
198 return;
199 case U'\b':
200 Backspace();
201 return;
202 }
203
204 // Write character at cursor position
205 if (m_cursorY >= 0 && m_cursorY < m_rows &&
206 m_cursorX >= 0 && m_cursorX < m_cols) {
207 int idx = CellIndex(m_cursorX, m_cursorY);
208 m_buffer[idx].ch = ch;
209 m_buffer[idx].attr = attr;
210 m_dirty = true;
211 }
212
213 // Advance cursor
214 m_cursorX++;
215
216 // Wrap to next line if needed
217 if (m_cursorX >= m_cols) {
218 m_cursorX = 0;
219 m_cursorY++;
220
221 // Mark the new line as soft-wrapped (continuation of previous line)
222 if (m_cursorY < m_rows && m_cursorY < static_cast<int>(m_lineWrapped.size())) {
223 m_lineWrapped[m_cursorY] = true;
224 }
225
226 // Scroll if at bottom of scroll region
227 int bottom = GetScrollRegionBottom();
228 if (m_cursorY > bottom) {
229 // Check if an explicit scroll region is set (m_scrollBottom >= 0)
230 bool hasExplicitScrollRegion = (m_scrollBottom >= 0);
231
232 if (hasExplicitScrollRegion) {
233 // Explicit scroll region - scroll within region, discard top
234 ScrollRegionUp(1);
235 } else {
236 // No explicit scroll region - scroll whole screen, add to scrollback
237 ScrollUp(1);
238 }
239 m_cursorY = bottom;
240 }
241 }

250void ScreenBuffer::ScrollUp(int lines) {
251 if (lines <= 0) return;
252
253 lines = std::min(lines, m_rows);
254
255 // Log scroll events to track resize issues
256 spdlog::info("ScrollUp({}) called, altBuf={}, scrollTop={}, scrollBottom={}",
257 lines, m_useAltBuffer ? "yes" : "no", m_scrollTop, GetScrollRegionBottom());
258
259 // Don't add to scrollback if using alt buffer
260 if (!m_useAltBuffer) {
261 // Move top lines to scrollback
262 for (int i = 0; i < lines; ++i) {
263 if (m_scrollbackUsed < m_scrollbackLines) {
264 // Add to end of scrollback
265 for (int x = 0; x < m_cols; ++x) {
266 m_scrollback.push_back(m_buffer[i * m_cols + x]);
267 }
268 m_scrollbackUsed++;
269 } else {
270 // Scrollback full, shift and replace last line
271 std::move(m_scrollback.begin() + m_cols,
272 m_scrollback.end(),
273 m_scrollback.begin());
274
275 // Replace last line
276 for (int x = 0; x < m_cols; ++x) {
277 m_scrollback[(m_scrollbackUsed - 1) * m_cols + x] =
278 m_buffer[i * m_cols + x];
279 }
280 }
281 }
282 }
283
284 // Shift buffer up
285 std::move(m_buffer.begin() + lines * m_cols,
286 m_buffer.end(),
287 m_buffer.begin());
288

290 std::fill(m_buffer.end() - lines * m_cols,
291 m_buffer.end(),
292 Cell());

296 std::move(m_lineWrapped.begin() + lines,
297 m_lineWrapped.end(),

661uint16_t ScreenBuffer::AddHyperlink(const std::string& url, const std::string& id) {
662 if (url.empty()) {
663 // Empty URL means end the current hyperlink
664 m_currentHyperlinkId = 0;
665 return 0;
666 }
667
668 // Check if we already have a hyperlink with the same URL and ID
669 for (const auto& [existingId, link] : m_hyperlinks) {
670 if (link.url == url && link.id == id) {
671 m_currentHyperlinkId = existingId;
672 return existingId;
673 }
674 }
675
676 // Create a new hyperlink entry
677 uint16_t newId = m_nextHyperlinkId++;
678 m_hyperlinks[newId] = Hyperlink{url, id};
679 m_currentHyperlinkId = newId;
680
681 // Update current attributes to include hyperlink flag
682 m_currentAttr.flags2 |= CellAttributes::FLAG2_HYPERLINK;
683 m_currentAttr.hyperlinkId = newId;
684
685 return newId;
686}
